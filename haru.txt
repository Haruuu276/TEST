local function glitchEffect(label)
    coroutine.wrap(function()
        local original = label.TextColor3
        for _ = 1, 2 do
            label.TextColor3 = Color3.new(1, 0, 0)
            wait(0.07)
            label.TextColor3 = original
            wait(0.07)
        end
    end)()
end

local function flash(button)
    local original = button.BackgroundColor3
    for _ = 1, 3 do
        button.BackgroundColor3 = Color3.new(1, 1, 1)
        wait(0.05)
        button.BackgroundColor3 = original
        wait(0.05)
    end
end

local function createESP(model, petName)
    local base = model:FindFirstChildWhichIsA("BasePart")
    if not base then return end

    -- Cleanup old
    for _, child in ipairs(model:GetChildren()) do
        if child.Name == "PetBillboard" or child.Name == "ESPHighlight" then
            child:Destroy()
        end
    end
    if not ESP_ENABLED then return end

    -- Status check
    local isReady = true
    local hatchTime = model:FindFirstChild("HatchTime")
    local readyFlag = model:FindFirstChild("ReadyToHatch")

    if (hatchTime and hatchTime.Value > 0) or (readyFlag and not readyFlag.Value) then
        isReady = false
    end

    -- Billboard
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "PetBillboard"
    billboard.Size = UDim2.new(0, 270, 0, 50)
    billboard.StudsOffset = Vector3.new(0, 4.5, 0)
    billboard.AlwaysOnTop = true
    billboard.MaxDistance = 500
    billboard.Parent = base

    local label = Instance.new("TextLabel", billboard)
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Font = Enum.Font.FredokaOne
    label.TextScaled = true
    label.TextStrokeTransparency = isReady and 0 or 0.5
    label.TextColor3 = isReady and Color3.new(1,1,1) or Color3.fromRGB(160,160,160)
    label.Text = model.Name .. " | " .. petName .. (isReady and "" or " (Not Ready)")

    glitchEffect(label)

    -- Highlight
    local highlight = Instance.new("Highlight", model)
    highlight.Name = "ESPHighlight"
    highlight.Adornee = model
    highlight.FillColor = Color3.fromRGB(255, 200, 0)
    highlight.OutlineColor = Color3.new(1,1,1)
    highlight.FillTransparency = 0.7
    highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
end

local function removeESP(model)
    for _, obj in ipairs(model:GetChildren()) do
        if obj.Name == "PetBillboard" or obj.Name == "ESPHighlight" then
            obj:Destroy()
        end
    end
end

local function getNearbyEggs(radius)
    local eggs = {}
    local root = (player.Character or player.CharacterAdded:Wait()):FindFirstChild("HumanoidRootPart")
    if not root then return eggs end

    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("Model") and EggPets[obj.Name] then
            local dist = (obj:GetModelCFrame().Position - root.Position).Magnitude
            if dist <= (radius or 60) then
                if not PetMemory[obj] then
                    local petList = EggPets[obj.Name]
                    PetMemory[obj] = petList[math.random(#petList)]
                end
                table.insert(eggs, obj)
            end
        end
    end
    return eggs
end

local function randomizeEggs()
    local eggs = getNearbyEggs()
    for _, egg in ipairs(eggs) do
        local petList = EggPets[egg.Name]
        local pet = petList[math.random(#petList)]
        PetMemory[egg] = pet
        createESP(egg, pet)
    end
    print("Randomized", #eggs, "eggs.")
end
